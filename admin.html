-- 1. Site content (hero, stats, contact info)
CREATE TABLE site_content (
  id TEXT PRIMARY KEY,
  value JSONB NOT NULL,
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 2. Gallery images
CREATE TABLE gallery (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  image_url TEXT NOT NULL,
  category TEXT,
  bride_name TEXT,
  quote TEXT,
  badge TEXT DEFAULT 'Our Work',
  sort_order INT DEFAULT 0,
  is_active BOOLEAN DEFAULT TRUE,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 3. Testimonials / Reviews
CREATE TABLE testimonials (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  client_name TEXT NOT NULL,
  initials TEXT,
  occasion TEXT,
  review_text TEXT NOT NULL,
  stars INT DEFAULT 5,
  avatar_color TEXT DEFAULT 'rose',
  sort_order INT DEFAULT 0,
  is_active BOOLEAN DEFAULT TRUE,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 4. FAQ items
CREATE TABLE faq (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  category TEXT NOT NULL,
  question TEXT NOT NULL,
  answer TEXT NOT NULL,
  sort_order INT DEFAULT 0,
  is_active BOOLEAN DEFAULT TRUE,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 5. Bridal packages
CREATE TABLE bridal_packages (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  name TEXT NOT NULL,
  tagline TEXT,
  icon TEXT,
  features JSONB DEFAULT '[]',
  is_featured BOOLEAN DEFAULT FALSE,
  sort_order INT DEFAULT 0
);

-- 6. Journey timeline
CREATE TABLE timeline (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  icon TEXT,
  year_label TEXT,
  tag TEXT,
  heading TEXT NOT NULL,
  body TEXT,
  sort_order INT DEFAULT 0
);

-- 7. Skin care treatments
CREATE TABLE treatments (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  name TEXT NOT NULL,
  tagline TEXT,
  icon TEXT,
  description TEXT,
  includes JSONB DEFAULT '[]',
  sort_order INT DEFAULT 0
);

-- Enable Row Level Security (RLS)
ALTER TABLE site_content ENABLE ROW LEVEL SECURITY;
ALTER TABLE gallery ENABLE ROW LEVEL SECURITY;
ALTER TABLE testimonials ENABLE ROW LEVEL SECURITY;
ALTER TABLE faq ENABLE ROW LEVEL SECURITY;
ALTER TABLE bridal_packages ENABLE ROW LEVEL SECURITY;
ALTER TABLE timeline ENABLE ROW LEVEL SECURITY;
ALTER TABLE treatments ENABLE ROW LEVEL SECURITY;

-- Allow public READ (your website visitors)

CREATE POLICY "Public read" ON gallery FOR SELECT USING (TRUE);
CREATE POLICY "Public read" ON testimonials FOR SELECT USING (TRUE);
CREATE POLICY "Public read" ON faq FOR SELECT USING (TRUE);
CREATE POLICY "Public read" ON bridal_packages FOR SELECT USING (TRUE);
CREATE POLICY "Public read" ON timeline FOR SELECT USING (TRUE);
CREATE POLICY "Public read" ON treatments FOR SELECT USING (TRUE);

-- Allow anon key to WRITE (admin panel uses anon key)
-- NOTE: For production, use service_role key only in admin
CREATE POLICY "Anon write" ON site_content FOR ALL USING (TRUE);
CREATE POLICY "Anon write" ON gallery FOR ALL USING (TRUE);
CREATE POLICY "Anon write" ON testimonials FOR ALL USING (TRUE);
CREATE POLICY "Anon write" ON faq FOR ALL USING (TRUE);
CREATE POLICY "Anon write" ON bridal_packages FOR ALL USING (TRUE);
CREATE POLICY "Anon write" ON timeline FOR ALL USING (TRUE);
CREATE POLICY "Anon write" ON treatments FOR ALL USING (TRUE);

-- Create Storage bucket for images
INSERT INTO storage.buckets (id, name, public)

CREATE POLICY "Public gallery read" ON storage.objects

CREATE POLICY "Anon gallery upload" ON storage.objects

CREATE POLICY "Anon gallery delete" ON storage.objects 
FOR DELETE USING (bucket_id = 'gallery');

4
Add Loader Script to Your Website Pages
Pending ▼
ℹ️Add this one script tag to the <head> of each HTML page. It will automatically load content from Supabase and update the page.

Copy your URL and key from Connect Settings first, then copy this script:

<!-- Supabase Dynamic Loader — Glow Pavi's Makeover -->
<script>
const SUPA_URL = 'YOUR_SUPABASE_URL';   // Replace with your URL
const SUPA_KEY = 'YOUR_ANON_KEY';       // Replace with your anon key

async function supaFetch(table, query='') {
  const res = await fetch(`${SUPA_URL}/rest/v1/${table}${query}`, {
    headers: {
      'apikey': SUPA_KEY,
      'Authorization': `Bearer ${SUPA_KEY}`,
      'Content-Type': 'application/json'
    }
  });
  return res.json();
}

async function loadSiteContent(key) {
  const data = await supaFetch('site_content', `?id=eq.${key}`);
  return data[0]?.value;
}

async function initPage() {
  // Load hero content
  const hero = await loadSiteContent('hero');
  if (hero) {
    const eyebrow = document.querySelector('.hero-eyebrow span');
    if (eyebrow) eyebrow.textContent = hero.eyebrow || eyebrow.textContent;
    const t1 = document.querySelector('.hero-title');
    if (t1 && hero.title1) t1.innerHTML = `${hero.title1}<br><em>${hero.title2}</em>`;
    const sub = document.querySelector('.hero-subtitle');
    if (sub && hero.subtitle) sub.textContent = hero.subtitle;
  }

  // Load stats
  const stats = await loadSiteContent('stats');
  if (stats) {
    const nums = document.querySelectorAll('.stat-num[data-target]');
    const vals = [stats.stat1, stats.stat2, stats.stat3, stats.stat4];
    nums.forEach((el, i) => {
      if (vals[i]) {
        el.dataset.target = vals[i].num;
        el.dataset.suffix = vals[i].suffix;
      }
    });
    const lbls = document.querySelectorAll('.stat-label');
    const labels = [stats.stat1?.label, stats.stat2?.label, stats.stat3?.label, stats.stat4?.label];
    lbls.forEach((el, i) => { if (labels[i]) el.textContent = labels[i]; });
  }

  // Load contact / WhatsApp links
  const contact = await loadSiteContent('contact');
  if (contact?.whatsapp) {
    document.querySelectorAll('a[href*="wa.me"]').forEach(a => {
      a.href = `https://wa.me/${contact.whatsapp}`;
    });
  }

  // Load gallery (on index.html)
  const galleryGrid = document.getElementById('galleryGrid') || document.querySelector('.gallery-grid');
  if (galleryGrid) {
    const images = await supaFetch('gallery', '?is_active=eq.true&order=sort_order');
    if (images?.length) renderGallery(galleryGrid, images);
  }

  // Load testimonials
  const tTrack = document.getElementById('testimonialsTrack');
  if (tTrack) {
    const reviews = await supaFetch('testimonials', '?is_active=eq.true&order=sort_order');
    if (reviews?.length) renderTestimonials(tTrack, reviews);
  }
}

function renderGallery(grid, images) {
  grid.innerHTML = images.map((img, i) => `
    <div class="gallery-item ${i===0?'tall':i===3?'wide':''} reveal" style="transition-delay:${i*0.07}s">
      <div class="gallery-fill" style="background:url('${img.image_url}') center/cover"></div>
      <div class="gallery-shine"></div>
      <div class="gallery-overlay"></div>
      <div class="gallery-static"><div class="gallery-dot"></div><span class="gallery-static-name">${img.category||''}</span></div>
      <div class="gallery-caption">
        <span class="gallery-badge">✦ ${img.badge||'Our Work'}</span>
        <p class="gallery-name">${img.bride_name||''}</p>
        <p class="gallery-quote">"${img.quote||''}"</p>
      </div>
    </div>`).join('');
}

function renderTestimonials(track, reviews) {
  const makeCard = r => `
    <div class="review-card">
      <div class="review-quote-mark">"</div>
      <div class="review-stars">${'<span class="star">★</span>'.repeat(r.stars||5)}</div>
      <p class="review-text">"${r.review_text}"</p>
      <div class="review-author">
        <div class="author-avatar ${r.avatar_color||'rose'}">${r.initials||'?'}</div>
        <div><p class="author-name">${r.client_name}</p><p class="author-type">${r.occasion||''}</p></div>
      </div>
    </div>`;
  const all = [...reviews,...reviews,...reviews];
  track.innerHTML = all.map(makeCard).join('');
}

document.addEventListener('DOMContentLoaded', initPage);
</script>
